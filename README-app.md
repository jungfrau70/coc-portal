# 8. Configure nuxt.config.js  -- <head> element
#############################################################################################

# install dependencies
yarn add @nuxtjs/vuetify @nuxtjs/axios

# configure Nuxt Env.
cat <<EOF > nuxt.config.js
export default {
    buildModules: [
        "@nuxtjs/vuetify"
    ],
    modules: [
        "@nuxtjs/axios"
    ],
    components: true
}
EOF

#############################################################################################
# 9. coding in pages/index.vue
#############################################################################################

cat <<EOF > pages/index.vue
<template lang="">
    <v-row>
        <v-col>
            <Form />
        </v-col>
        <v-col>
            <Table />
        </v-col>
    </v-row>
</template>
EOF
